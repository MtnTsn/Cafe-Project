CREATE PROC SP_GarsonGuncelleme


(@adi varchar(50),
@sifre varchar(32),
@id int,
@tel varchar(100),
@mail varchar(255),
@adres varchar(500)

)
AS

BEGIN

UPDATE kullanicilar   SET kulAdi=@adi,sifre=@sifre  WHERE profilID=@id 
Update profil  SET telefon=@tel, mail=@mail, adres=@adres WHERE  id=@id

END
